var tools,
    incr;
/**
 * @namespace
 * @private

tools = {
    /**
     * a = b ?
     *
     * @param {*} a
     * @param {*} b
     *
     * @returns {boolean}
     */
    isEqual: (a, b) => (a == b),
    /**
     * a =/= b ?
     *
     * @param {*} a
     * @param {*} b
     *
     * @returns {boolean}
     */
    isNotEqual: (a, b) => (a != b),
    /**
     * a has b ?
     *
     * @param {*} a
     * @param {*} b
     *
     * @returns {boolean}
     */

    /**
     * a dont have b ?
     *
     * @param {*} a
     * @param {*} b
     *
     * @returns {boolean}
     */
    notContains: (a, b) => (!(b in a)),
    /**
     * type of a = type of b ?
     *
     * @param {*} a
     * @param {*} b
     *
     * @returns {boolean}

/**
 * Tester - Creates a test sequence
 */
class Tester {
    /**
     * @param  {(string|number)} name A name to make a contrast between tests
     * @param  {function} func The function which going to be runned
     */
    constructor(name = `Anm${++incr}`, func) {
        this.name = name.toUpperCase();
        this.func = func;
    }

    /**
     * onerror - The function to set handler to errors
     *
     * @param  {function} func The function to set as handler
     * @returns {object} The Main class will return
     */
    onerror(func) {
        this.error = func;
        return this;
    };

    /**
     * onsuccess - The function to set handler to success
     *
     * @param  {function} func The function to set as handler
     * @returns {object} The Main class will return
     */
    onsuccess(func) {
        this.success = func;
        return this;
    };

    /**
     * run - Will run the test
     *
     * @returns {boolean} Will say if all resualts are true
     */
    run() {
        var out = this.func(tools, new CustomArray()),
            outlen = out.length;
        for (var i = 0; i < outlen; i++)
            if (out[i].status == true)
                this.success.bind(this, out[i].desc)();
            else
                this.error.bind(this, out[i].desc)();
        return (out.map(a => (a.status == false)).indexOf(true) == -1);
    };
}

/**
 * @extends Array
 * @private
 */
class CustomArray extends Array {
    /**
     * add - Addes a option to array
     *
     * @param {boolean} status
     * @param {string} desc [REST]
     *
     * @returns {object} The object it self
     */
    add(status, ...desc) {
        this.push({
            status,
            desc: desc.join("\n")
        });
        return this;
    };
}

module.exports = Tester;
